    <script>
        // Paste your OpenWeatherMap API Key here
        const apiKey = "YOUR_API_KEY_HERE"; 
        const apiUrl = "https://api.openweathermap.org/data/2.5/weather?units=metric&q=";

        const cityInput = document.getElementById("city-input");
        const searchBtn = document.getElementById("search-btn");
        const weatherBox = document.getElementById("weather-box");
        const errorMsg = document.getElementById("error-msg");

        async function checkWeather(city) {
            if (apiKey === "YOUR_API_KEY_HERE") {
                alert("Please add your OpenWeatherMap API Key in the script block!");
                return;
            }

            try {
                const response = await fetch(apiUrl + city + `&appid=${apiKey}`);
                
                if (response.status == 404) {
                    errorMsg.style.display = "block";
                    weatherBox.classList.remove("active");
                } else {
                    const data = await response.json();

                    document.getElementById("city-name").innerText = data.name;
                    document.getElementById("temperature").innerText = Math.round(data.main.temp) + "°C";
                    document.getElementById("weather-desc").innerText = data.weather[0].description;
                    document.getElementById("humidity-val").innerText = data.main.humidity + "%";
                    document.getElementById("wind-val").innerText = data.wind.speed + " km/h";

                    weatherBox.classList.add("active");
                    errorMsg.style.display = "none";
                }
            } catch (error) {
                console.error("Error fetching data:", error);
            }
        }

        searchBtn.addEventListener("click", () => {
            if(cityInput.value.trim() !== "") {
                checkWeather(cityInput.value);
            }
        });

        // Allow pressing 'Enter' key to search
        cityInput.addEventListener("keydown", (event) => {
            if (event.key === "Enter" && cityInput.value.trim() !== "") {
                checkWeather(cityInput.value);
            }
        });
    </script>
